import cv2
import os
# Load Haarcascade
face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + "haarcascade_frontalface_default.xml")
# Create dataset directory
dataset_path = "dataset"
os.makedirs(dataset_path, exist_ok=True)
# Enter person name
person_name = input("Enter your name: ").strip()
person_path = os.path.join(dataset_path, person_name)
os.makedirs(person_path, exist_ok=True)
cap = cv2.VideoCapture(0)
count = 0
print("[INFO] Capturing faces... Look at the camera.")
while True:
ret, frame = cap.read()
if not ret:
break
gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
faces = face_cascade.detectMultiScale(gray, 1.3, 5)
for (x, y, w, h) in faces:
count += 1
face = gray[y:y+h, x:x+w]
face = cv2.resize(face, (200, 200))
cv2.imwrite(f"{person_path}/{count}.jpg", face)
20
print(f"[INFO] Saved image {count}")
# stop after 50 images
if count >= 200:
break
cap.release()
print("[INFO] Dataset collection complete!")
