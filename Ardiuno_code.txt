import cv2
import pickle
import serial
import time
# Setup serial communication with Arduino (adjust COM port & baud rate)
arduino = serial.Serial("COM8", 9600, timeout=1)
time.sleep(2) # wait for Arduino reset
# Load cascade & trained model
face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + "haarcascade_frontalface_default.xml")
recognizer = cv2.face.LBPHFaceRecognizer_create()
recognizer.read("face_model.yml")
with open("labels.pkl", "rb") as f:
label_dict = pickle.load(f)
cap = cv2.VideoCapture(0)
print("[INFO] Starting face recognition... Press ESC to exit.")
while True:
ret, frame = cap.read()
if not ret:
break 22
gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
faces = face_cascade.detectMultiScale(gray, 1.3, 5)
for (x, y, w, h) in faces:
roi_gray = gray[y:y+h, x:x+w]
roi_gray = cv2.resize(roi_gray, (200, 200))
roi_gray = cv2.equalizeHist(roi_gray)
label, confidence = recognizer.predict(roi_gray)
if confidence < 50: # Good match
name = label_dict[label]
text = f"{name} ({int(confidence)})"
color = (0, 255, 0)
# Send unique code to Arduino for each person
if name == "abishek":
arduino.write(b"A\n")
print ("abishek")
elif name == "abinesh":
arduino.write(b"B\n")
print ("abinesh")
elif name == "aadithya":
arduino.write(b"C\n")
print ("aadithya")
elif name == "sharon":
arduino.write(b"D\n")
print ("sharon")
else:
text = "Unknown"
color = (0, 0, 255)
arduino.write(b"X\n") # Unknown person
cv2.putText(frame, text, (x, y-10),
cv2.FONT_HERSHEY_SIMPLEX, 0.9, color, 2) 23
cv2.rectangle(frame, (x, y), (x+w, y+h), color, 2)
cv2.imshow("Face Recognition", frame)
if cv2.waitKey(1) == 27: # ESC to quit
break
cap.release()
cv2.destroyAllWindows()
arduino.close()
